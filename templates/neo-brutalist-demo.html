Neo Brutalist Component Library

Button

NeoBrutalist.create('Button', '#container', {
  text: 'Click Me',
  color: 'blue-500',
  onClick: () => alert('Clicked!')
});

NeoBrutalist.create('Card', '#container', {

});

});
